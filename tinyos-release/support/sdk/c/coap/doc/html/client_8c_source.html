00412 int
00413 join( coap_context_t *ctx, char *group_name ){
00414   struct ipv6_mreq mreq;
00415   struct addrinfo   *reslocal = NULL, *resmulti = NULL, hints, *ainfo;
00416   int result = -1;
00417 
00418   /* we have to resolve the link-local interface to get the interface id */
00419   memset(&hints, 0, sizeof(hints));
00420   hints.ai_family = AF_INET6;
00421   hints.ai_socktype = SOCK_DGRAM;
00422 
00423   result = getaddrinfo("::", NULL, &hints, &reslocal);
00424   if ( result < 0 ) {
00425     perror("join: cannot resolve link-local interface");
00426     goto finish;
00427   }
00428 
00429   /* get the first suitable interface identifier */
00430   for (ainfo = reslocal; ainfo != NULL; ainfo = ainfo->ai_next) {
00431     if ( ainfo->ai_family == AF_INET6 ) {
00432       mreq.ipv6mr_interface =
00433     ((struct sockaddr_in6 *)ainfo->ai_addr)->sin6_scope_id;
00434       break;
00435     }
00436   }
00437 
00438   memset(&hints, 0, sizeof(hints));
00439   hints.ai_family = AF_INET6;
00440   hints.ai_socktype = SOCK_DGRAM;
00441 
00442   /* resolve the multicast group address */
00443   result = getaddrinfo(group_name, NULL, &hints, &resmulti);
00444 
00445   if ( result < 0 ) {
00446     perror("join: cannot resolve multicast address");
00447     goto finish;
00448   }
00449 
00450   for (ainfo = resmulti; ainfo != NULL; ainfo = ainfo->ai_next) {
00451     if ( ainfo->ai_family == AF_INET6 ) {
00452       mreq.ipv6mr_multiaddr =
00453     ((struct sockaddr_in6 *)ainfo->ai_addr)->sin6_addr;
00454       break;
00455     }
00456   }
00457 
00458   result = setsockopt( ctx->sockfd, IPPROTO_IPV6, IPV6_JOIN_GROUP,
00459                (char *)&mreq, sizeof(mreq) );
00460   if ( result < 0 )
00461     perror("join: setsockopt");
00462 
00463  finish:
00464   freeaddrinfo(resmulti);
00465   freeaddrinfo(reslocal);
