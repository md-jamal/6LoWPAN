00145          int (*order)(coap_queue_t *, coap_queue_t *node) ) {
00146   coap_queue_t *p, *q;
00147   if ( !queue || !node )
00148     return 0;
00149     
00150   /* set queue head if empty */
00151   if ( !*queue ) {
00152     *queue = node;
00153     return 1;
00154   }
00155 
00156   /* replace queue head if PDU's time is less than head's time */
00157   q = *queue;
00158   if ( order( node, q ) < 0) {
00159     node->next = q;
00160     *queue = node;
00161     return 1;
00162   }
00163 
00164   /* search for right place to insert */
00165   do {
00166     p = q;
00167     q = q->next;
00168   } while ( q && order( node, q ) >= 0 );

00172   p->next = node;
